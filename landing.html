Dead simple to use

# Generate a screenshot
curl "https://api.example.com/screenshot/1617979122625712128" \
  -H "X-API-KEY: your-api-key" \
  -o tweet.png

# With options
curl "https://api.example.com/screenshot/1617979122625712128?theme=dark&gradient=ocean&dimension=instagramFeed" \
  -H "X-API-KEY: your-api-key" \
  -o tweet.png
